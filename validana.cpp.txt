class Solution {
public:
    bool isAnagram(string s, string t) {
        vector<int>count(256,0);
        if(s.size()!=t.size()){
            return false;
        }
        for(char c:s){
            count[c]++;
        }
        for(char c:t){
            count[c]--;
        }
        for(int i :count){
            if(i!=0)return false;
        }return true;
    }
};
