class Solution {
public:
    int maxProfit(vector<int>& prices, int fee) {
        int hold=-prices[0];
        int cash=0;
        for(int i=1;i<prices.size();i++){
            //hold has been used over here since we are buying 
            hold=max(hold,cash-prices[i]);
            //cash has been used over here to signify the selling process
            cash=max(cash,hold+prices[i]-fee);
        }return cash;
    }
};
